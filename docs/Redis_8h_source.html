236 Redis::TestResults Redis::runTests(bool logToSerial, String prefix, bool retainData)
237 {
238  /* These are NOT executed in definition order! */
239  std::map<String, std::function<bool(const char*)>> g_Tests
240  {
241  { "set", [this](const char* k) { return set(k, "!"); } },
242  { "setget", [this](const char* k) { return set(k, "!") == 1 && get(k) == "!"; } },
243  { "expire", [this](const char* k) {
244  return set(k, "E") && expire(k, 5) && ttl(k) > 4;
245  } },
246  { "pexpire", [this](const char* k) {
247  return set(k, "PE") && pexpire(k, 5000) && pttl(k) > 4500;
248  } },
249  { "dlyexp", [this](const char* k) {
250  auto sr = set(k, "DE");
251  expire(k, 10);
252  delay(4000);
253  auto t = ttl(k);
254  return sr && t > 5 && t < 7;
255  } },
256  { "hset", [this](const char* k) { return hset(k, "TEST-FIELD", "H!"); } },
257  { "hsetget", [this](const char* k) {
258  return hset(k, "TF", "HH") && hget(k, "TF") == "HH";
259  } },
260  { "hlen", [this](const char* k) {
261  for (int i = 0; i < 10; i++) {
262  auto fv = String(i);
263  hset(k, String("field-" + fv).c_str(), fv.c_str());
264  }
265  return hlen(k) == 10;
266  } },
267  { "hstrlen", [this](const char* k) {
268  return hset(k, "hsr", k) && hstrlen(k, "hsr") == strlen(k);
269  } },
270  { "append", [this](const char* k) { return append(k, "foo") == 3; } },
271  { "exists", [this](const char* k) { return set(k, k) && exists(k); } }
272  };
273 
274  if (logToSerial) {
275  Serial.begin(115200);
276  Serial.printf("Redis::run_tests(\"%s\") starting...\n", prefix.c_str());
277  }
278 
279  auto pFunc = [&prefix](const String& n) { return String(prefix + "." + n); };
280 
281  int total = 0, pass = 0;
282  for (auto& kv : g_Tests)
283  {
284  auto tName = kv.first;
285  auto tFunc = kv.second;
286  auto pName = pFunc(tName);
287  auto tRes = tFunc(pName.c_str());
288  if (logToSerial)
289  Serial.printf("\tTest \"%s\":\t%s\n", tName.c_str(), tRes ? "pass" : "FAIL");
290  total++;
291  pass += tRes ? 1 : 0;
292  }
293 
294  if (logToSerial)
295  Serial.printf("Result: %d passed / %d total\n", pass, total);
296 
297  if (!retainData)
298  for (auto& kv : g_Tests)
299  (void)del(pFunc(kv.first).c_str());
300 
301  return { .total = total, .passed = pass };
302 }
